84       public static void start_mr(String in, String out) throws IOException, ClassNotFoundException, InterruptedException {
85           Configuration conf = new Configuration();
86           Job job = Job.getInstance(conf, "MapReduce");
87           job.setJarByClass(Hadoop.class);
88           job.setMapperClass(Hadoop.TokenizerMapper.class);
89           job.setNumReduceTasks(1);
90           job.setReducerClass(Hadoop.IntSumReducer.class);
91           job.setOutputKeyClass(Text.class);
92           job.setOutputValueClass(Text.class);
93           FileInputFormat.addInputPath(job, new Path(in));
94           FileOutputFormat.setOutputPath(job, new Path(out));
95           FileSystem.get(conf).delete(new Path(out), true);
96           job.waitForCompletion(true);
